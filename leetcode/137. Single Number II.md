class Solution {
    public int singleNumber(int[] nums) {
        int N = nums.length;
        int ans = 0;
        for(int i=0; i < 32; i++){
            int count = 0;
            for(int j=0; j < N; j++){
                if(checkBit(nums[j], i)){
                    count++;
                }
            }
            if(count%3 == 1){
                ans = ans + (1<<i);
            }
        }
        return ans;
    }
    
    public boolean checkBit(int num, int i){
        return (num & (1 << i)) != 0;
    }
}
